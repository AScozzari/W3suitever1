// mcp-registry.service.ts
export type McpServer = {
  id: string;
  name: string;
  description?: string;
  homepage?: string;
  repo?: string;
  version?: string;
  verified?: boolean;     // dal registry (DNS/namespace)
  auth?: { type: 'none'|'apiKey'|'oauth2'; scopes?: string[] };
  install: { npm?: string; docker?: string; manual?: string };
  categories?: string[];
  publisher?: { name: string; url?: string };
};

const REGISTRY_API = 'https://registry.modelcontextprotocol.io/v1/servers/search?q='; // esempio

export async function fetchFromMcpRegistry(query = ''): Promise<McpServer[]> {
  const res = await fetch(`${REGISTRY_API}${encodeURIComponent(query)}`);
  if (!res.ok) return [];
  const data = await res.json();
  return data.results.map((s: any) => ({
    id: s.id,
    name: s.displayName ?? s.id,
    description: s.description,
    homepage: s.links?.homepage,
    repo: s.links?.repository,
    version: s.latestVersion,
    verified: !!s.verification?.dns,
    auth: s.auth,
    install: { npm: s.install?.npm, docker: s.install?.docker, manual: s.install?.manual },
    categories: s.categories,
    publisher: s.publisher,
  }));
}

// marketplace-merge.ts
import { fetchFromMcpRegistry } from './mcp-registry.service';
import { fetchFromCursorDir, fetchFromAwesomeMcp, fetchFromNpm } from './community-sources';
import { CURATED } from './curated-official';

export async function getMarketplace(query = '') {
  const [reg, cur, curDir, awesome, npm] = await Promise.allSettled([
    fetchFromMcpRegistry(query),
    Promise.resolve(CURATED),
    fetchFromCursorDir(query),
    fetchFromAwesomeMcp(query),
    fetchFromNpm(query), // opzionale
  ]);

  const all = [
    ...(cur.status === 'fulfilled' ? cur.value : []),
    ...(reg.status === 'fulfilled' ? reg.value : []),
    ...(curDir.status === 'fulfilled' ? curDir.value : []),
    ...(awesome.status === 'fulfilled' ? awesome.value : []),
    ...(npm.status === 'fulfilled' ? npm.value : []),
  ];

  // dedupe by id/repo/npm name + ranking
  const map = new Map<string, any>();
  for (const s of all) {
    const key = s.id ?? s.install?.npm ?? s.repo ?? s.name;
    const prev = map.get(key);
    map.set(key, rankBetter(prev, s)); // tieni la variante ‚Äúmigliore‚Äù (verified > curated > others)
  }
  return Array.from(map.values()).sort(sortByTrustAndPopularity);
}
